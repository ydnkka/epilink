% Muted scientific palette with a shared natural-history accent
\definecolor{BaseInk}{HTML}{22313B}        % Deep slate
\definecolor{SharedColor}{HTML}{5F8C78}    % Sage green
\definecolor{TemporalColor}{HTML}{B7793F}  % Burnished amber
\definecolor{GeneticColor}{HTML}{3C7E85}   % Ocean teal
\definecolor{ScenarioColor}{HTML}{8770B3}  % Muted iris
\definecolor{NeutralGrey}{HTML}{667685}    % Blue slate
\definecolor{BoxBg}{HTML}{F6F5F1}          % Warm paper

\tikzset{
  flow/.style={
    -{Latex[length=3.5mm,width=2.2mm]},
    draw=BaseInk,
    line width=1.1pt,
    shorten >= 2pt,
    shorten <= 2pt
  },
  scenario flow/.style={
    -{Latex[length=2.2mm,width=1.6mm]},
    draw=BaseInk,
    line width=0.8pt,
    shorten >= 1pt,
    shorten <= 1pt
  },
  workflow box/.style={
    rounded corners=5pt,
    very thick,
    rectangle split,
    rectangle split parts=2,
    font=\small,
    inner sep=8pt,
    align=left,
    text=BaseInk
  },
  neutral box/.style={
    workflow box,
    draw=NeutralGrey,
    rectangle split part fill={white, BoxBg}
  },
  shared box/.style={
    workflow box,
    draw=SharedColor,
    rectangle split part fill={SharedColor!12, white}
  },
  temporal box/.style={
    workflow box,
    draw=TemporalColor,
    rectangle split part fill={TemporalColor!12, white}
  },
  genetic box/.style={
    workflow box,
    draw=GeneticColor,
    rectangle split part fill={GeneticColor!12, white}
  },
  final box/.style={
    workflow box,
    draw=ScenarioColor,
    rectangle split part fill={ScenarioColor!15, white}
  },
  column box/.style={
    draw=ScenarioColor!45,
    dashed,
    line width=0.8pt,
    rounded corners=4pt,
    fill=white
  },
  sampled case/.style={
    circle, draw=BaseInk, fill=white, line width=1.1pt, minimum size=6.5mm, font=\scriptsize\bfseries
  },
  latent case/.style={
    sampled case, fill=NeutralGrey!15, dash pattern=on 2pt off 1pt
  }
}

\begin{document}
\begin{tikzpicture}[node distance=1.2cm and 0.8cm, font=\sffamily]

  % --- HEADER & INPUT ---
  \node[align=center, text=BaseInk] (title) at (8, 16) {\LARGE\bfseries EpiLink Compatibility Model};

  \node[neutral box, text width=6.5cm, below=0.6cm of title] (input) {%
    {\bfseries Observed Pairwise Data}
    \nodepart{second}
    Testing-time difference: $t_{ij}$ \\
    Consensus genetic distance: $g_{ij}$
  };

  % --- LATENT SCENARIOS CONTAINER ---
  \node[below=1cm of input, draw=ScenarioColor, fill=ScenarioColor!4, line width=1.5pt, rounded corners=8pt, inner sep=15pt, text width=12cm, minimum height=6.2cm] (scenarios_border) {};
  \node[anchor=north, font=\bfseries, text=ScenarioColor, yshift=-5pt] at (scenarios_border.north) {LATENT TRANSMISSION SCENARIOS ($\mathcal{S}_M$)};

  % Column 1: AD
  \node[column box, anchor=north west, minimum width=5.8cm, minimum height=5.1cm, xshift=10pt, yshift=-25pt] at (scenarios_border.north west) (ad_col) {};
  \node[anchor=north, font=\bfseries\small, text=BaseInk, yshift=-5pt] at (ad_col.north) {Ancestor-Descendant $H_{\mathrm{AD}}(m)$};
  \node[anchor=south, font=\small, text=BaseInk, yshift=2pt, text width=5cm, minimum width=6cm] at (ad_col.south) {Direct transmission $i \rightarrow j$, $m=0$.};

  \begin{scope}[node distance=0.7cm]
    \node[sampled case, below=1cm of ad_col.north] (adi) {$i$};
    \node[latent case, below=of adi] (adm) {$m$};
    \node[sampled case, below= of adm] (adj) {$j$};
    \draw[scenario flow] (adi) -- (adm);
    \draw[scenario flow] (adm) -- (adj);
  \end{scope}

  % Column 2: CA
  \node[column box, anchor=north east, minimum width=6.6cm, minimum height=5.1cm, xshift=-10pt, yshift=-25pt] at (scenarios_border.north east) (ca_col) {};
  \node[anchor=north, font=\bfseries\small, text=BaseInk, yshift=-5pt] at (ca_col.north) {Common Ancestor $H_{\mathrm{CA}}(m_i, m_j)$};
  \node[anchor=south, font=\small, text=BaseInk, yshift=2pt, text width=6cm, minimum width=5cm] at (ca_col.south) {Co-primary infection $(m_i, m_j)=(0,0)$.};

  \begin{scope}[node distance=0.5cm]
    \node[latent case, below=0.8cm of ca_col.north] (caa) {$a$};
    \node[latent case, below left=0.8cm and 0.8cm of caa] (cami) {$m_i$};
    \node[latent case, below right=0.8cm and 0.8cm of caa] (camj) {$m_i$};
    \node[sampled case, below=1cm of cami] (cai) {$i$};
    \node[sampled case, below=1cm of camj] (caj) {$j$};
    \draw[scenario flow] (caa) -- (cami);
    \draw[scenario flow] (caa) -- (camj);
    \draw[scenario flow] (cami) -- (cai);
    \draw[scenario flow] (camj) -- (caj);
  \end{scope}

  % --- SHARED NATURAL-HISTORY MODEL ---
  \node[shared box, below=1.1cm of scenarios_border.south, text width=6cm] (nh_model) {
    {\bfseries Natural-history Model}
    \nodepart{second}
    Gamma stages ($E/P/I$) \\
    \hspace{1em} Incubation period: $\tau_{\mathrm{inc}} = y_E + y_P$ \\
    \hspace{1em} Generation time: $\tau_{\mathrm{gen}} = \tau_{\mathrm{inc}} + y_{\mathrm{toit}}$ \\
    \hspace{1em} Testing delay: $x_{\mathrm{test}}$ \\
    Clock rate: $\lambda = rL/365$ (strict or relaxed)
  };

  % --- PARALLEL TEMPORAL AND GENETIC BRANCHES ---
  \node[temporal box, text width=8.9cm] (t_draws) at ($(nh_model.south)+(-4.9cm,-3.1cm)$) {
    {\bfseries MC Temporal Draws}
    \nodepart{second}
    AD: $T_s = \sum_{r=0}^{m}\tau_{\mathrm{gen},r} + \tau_{\mathrm{inc},j} + x_{\mathrm{test},j} - \left(\tau_{\mathrm{inc},i} + x_{\mathrm{test},i}\right)$ \\[4pt]
    CA: $T_s =\sum_{s=1}^{m_j+1}\tau_{\mathrm{gen},s}^{(j)} - \sum_{r=1}^{m_i+1}\tau_{\mathrm{gen},r}^{(i)} + \tau_{\mathrm{inc},j} + x_{\mathrm{test},j} - \left(\tau_{\mathrm{inc},i} + x_{\mathrm{test},i}\right)$ \\[4pt]
    \textit{Signed difference for temporal ordering}
  };

  \node[neutral box, below=1.2cm of t_draws, text width=5.2cm] (t_comp) {
    {\bfseries Temporal Compatibility}
    \nodepart{second}
    $C_{T,s} = 1 - 2|\Pr(T_s \le t_{ij}) - 0.5|$
  };

  % Genetic Track
  \node[genetic box, text width=8.9cm] (g_draws) at ($(nh_model.south)+(4.9cm,-3.1cm)$) {
    {\bfseries MC Genetic Draws}
    \nodepart{second}
    AD: $B_s = \sum \tau_{\mathrm{gen}} + \tau_{\mathrm{inc},j}+x_{\mathrm{test},j}-(\tau_{\mathrm{inc},i}+x_{\mathrm{test},i})$\\[4pt]
    CA: $B_s = \sum \tau_{\mathrm{gen}}^{(i)} + \sum \tau_{\mathrm{gen}}^{(j)}+\tau_{\mathrm{inc},i}+x_{\mathrm{test},i}+\tau_{\mathrm{inc},j}+x_{\mathrm{test},j}$ \\[4pt]
    Deterministic: $G_s = \lambda B_s$ or stochastic: $G_s \sim \operatorname{Poisson}(\lambda B_s)$
  };

  \node[neutral box, below=1.2cm of g_draws, text width=5.2cm] (g_comp) {
    {\bfseries Genetic Compatibility}
    \nodepart{second}
    $C_{G,s} = 1 - 2|\Pr(G_s \le g_{ij}) - 0.5|$
  };

  % --- FINAL SCORE (CENTER) ---
  \node[final box, text width=6.5cm] (final) at ($(t_comp.south)!0.5!(g_comp.south) + (0,-2.9cm)$) {
    {\bfseries Aggregated Compatibility Score}
    \nodepart{second}
    Scenario: $C_s(i,j) = C_{T,s} \cdot C_{G,s}$ \\
    Target Set: $C_{\mathcal{S}_\star} = \sum_{s \in \mathcal{S}_*} C_s(i,j)$
  };

  % --- CONNECTIONS ---
  \draw[flow] (input) -- (scenarios_border);
  \draw[flow] (scenarios_border.south) -- (nh_model.north);

  \coordinate (shared_split) at ($(nh_model.south)+(0,-0.9cm)$);
  \draw[flow] (nh_model.south) -- (shared_split);
  \draw[flow] (shared_split) -| (t_draws.north);
  \draw[flow] (shared_split) -| (g_draws.north);
  \draw[flow] (t_draws) -- (t_comp);
  \draw[flow] (g_draws) -- (g_comp);

  % Track converge to Final Score from West and East
  \draw[flow] (t_comp) |- (final);
  \draw[flow] (g_comp) |- (final);

\end{tikzpicture}
\end{document}
